const { Client, GatewayIntentBits } = require('discord.js');
const { DisTube } = require('distube');
const { YouTubePlugin } = require('@distube/youtube');
const { SpotifyPlugin } = require('@distube/spotify');

// إنشاء عميل الديسكورد مع الصلاحيات المطلوبة
const client = new Client({
    intents: [
        GatewayIntentBits.Guilds,
        GatewayIntentBits.GuildVoiceStates,
        GatewayIntentBits.GuildMessages,
        GatewayIntentBits.MessageContent
    ]
});

// إعداد مشغل الموسيقى DisTube مع دعم يوتيوب وسبوتيفاي
const distube = new DisTube(client, {
    emitNewSongOnly: true,
    leaveOnFinish: false,
    leaveOnStop: true,
    plugins: [
        new YouTubePlugin(),
        new SpotifyPlugin({ emitEventsAfterFetching: true })
    ]
});

const PREFIX = "!"; // بادئة الأوامر (يمكنك تغييرها)

client.once('ready', () => {
    console.log(`✅ تم تشغيل البوت بنجاح باسم: ${client.user.tag}`);
});

// معالجة الأوامر النصية
client.on('messageCreate', async (message) => {
    if (message.author.bot || !message.guild || !message.content.startsWith(PREFIX)) return;

    const args = message.content.slice(PREFIX.length).trim().split(/ +/g);
    const command = args.shift().toLowerCase();

    // أمر التشغيل (!play أو !p)
    if (command === 'play' || command === 'p') {
        const voiceChannel = message.member.voice.channel;
        if (!voiceChannel) {
            return message.reply('❌ يجب أن تكون في روم صوتي أولاً!');
        }

        const query = args.join(' ');
        if (!query) {
            return message.reply('❌ اكتب اسم الأغنية أو الرابط (يوتيوب/سبوتيفاي) بعد الأمر.');
        }

        try {
            message.reply(`🔍 جاري البحث والتشغيل: **${query}**...`);
            await distube.play(voiceChannel, query, {
                textChannel: message.channel,
                member: message.member
            });
        } catch (error) {
            console.error(error);
            message.channel.send('❌ حدث خطأ أثناء محاولة تشغيل الأغنية.');
        }
    }

    // أمر إيقاف الموسيقى مؤقتاً (!pause)
    if (command === 'pause') {
        const queue = distube.getQueue(message);
        if (!queue) return message.reply('❌ لا توجد موسيقى تعمل حالياً.');
        if (queue.paused) return message.reply('⏸️ الموسيقى متوقفة بالفعل.');
        distube.pause(message);
        message.reply('⏸️ تم إيقاف الموسيقى مؤقتاً.');
    }

    // أمر استئناف الموسيقى (!resume)
    if (command === 'resume') {
        const queue = distube.getQueue(message);
        if (!queue) return message.reply('❌ لا توجد موسيقى تعمل حالياً.');
        if (!queue.paused) return message.reply('▶️ الموسيقى تعمل بالفعل.');
        distube.resume(message);
        message.reply('▶️ تم استئناف الموسيقى.');
    }

    // أمر تخطي الأغنية (!skip)
    if (command === 'skip' || command === 's') {
        const queue = distube.getQueue(message);
        if (!queue) return message.reply('❌ لا توجد موسيقى في قائمة الانتظار للروم.');
        try {
            const song = await distube.skip(message);
            message.reply(`⏭️ تم التخطي إلى الأغنية التالية: **${song.name}**`);
        } catch (e) {
            message.reply('❌ لا توجد أغنية تالية لتخطيها (أو قائمة الانتظار فارغة).');
        }
    }

    // أمر إيقاف البوت وخروجه من الروم (!stop)
    if (command === 'stop' || command === 'leave') {
        const queue = distube.getQueue(message);
        if (!queue) return message.reply('❌ البوت ليس في روم صوتي حالياً.');
        distube.stop(message);
        message.reply('🛑 تم إيقاف التشغيل ومغادرة الروم.');
    }
});

// أحداث مشغل الموسيقى (إرسال رسائل تلقائية في الشات)
distube
    .on('playSong', (queue, song) => {
        queue.textChannel.send(`🎶 جاري تشغيل الآن: **${song.name}** - \`${song.formattedDuration}\`\nطلب بواسطة: ${song.user}`);
    })
    .on('addSong', (queue, song) => {
        queue.textChannel.send(`✅ تم إضافة **${song.name}** إلى قائمة الانتظار بواسطة ${song.user}`);
    })
    .on('error', (channel, e) => {
        console.error(e);
        channel.send(`❌ واجهت مشكلة: ${e.toString().slice(0, 100)}`);
    });

// تشغيل البوت باستخدام التوكن المخزن في البيئة الحمائية
client.login(process.env.TOKEN);
